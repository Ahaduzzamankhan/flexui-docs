TypeScript

FlexUI ships with full TypeScript type definitions. No @types package needed — types are included in the main package.

Project Setup

Scaffold a TypeScript project with:

flexui create my-app --ts

Or add TypeScript manually with the recommended tsconfig.json:

tsconfig.json 
{
  "compilerOptions": {
    "target":           "ES2020",
    "module":           "ESNext",
    "moduleResolution": "Bundler",
    "strict":           true,
    "jsx":              "react-jsx",
    "jsxImportSource":  "flexui",
    "types":            ["flexui"]
  },
  "include": ["src", "pages"]
}

Component Types

import type { FC, FlexElement } from 'flexui';

// FC<Props> — functional component shorthand
interface ButtonProps {
  label:    string;
  onClick?: () => void;
  variant?: 'primary' | 'secondary';
}

const Button: FC<ButtonProps> = ({ label, onClick, variant = 'primary' }) => {
  return createElement('button', { className: `btn--${variant}`, onClick }, label);
};

Hook Types

import { useState, useRef, useReducer } from 'flexui';

// useState infers type from initial value
const [count, setCount] = useState(0);
// count: number

// Explicit type when initial value is null
const [user, setUser] = useState<User | null>(null);

// useRef
const inputRef = useRef<HTMLInputElement>(null);

// useReducer
type Action = { type: 'inc' } | { type: 'reset' };
const [state, dispatch] = useReducer<State, Action>(reducer, { count: 0 });

Router Types

import type { RouterState } from 'flexui/router';

// RouterState shape
interface RouterState {
  pathname: string;
  query:    Record<string, string>;
  params:   Record<string, string>;
  push:     (url: string) => void;
  replace:  (url: string) => void;
  back:     () => void;
}

Config Types

import type { FlexUIConfig } from 'flexui';
import { defineConfig } from 'flexui/build';

const config: FlexUIConfig = defineConfig({
  entry:   'src/main.ts',
  port:    3000,
  minify:  true,
});
