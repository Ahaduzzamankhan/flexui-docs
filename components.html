Functional Components

A function that accepts props and returns a FlexElement. This is the recommended way to build UI in FlexUI.

components/Button.js 
import { createElement } from 'flexui';

export function Button({ label, onClick, variant = 'primary' }) {
  return createElement(
    'button',
    { className: `btn btn--${variant}`, onClick },
    label
  );
}

Class Components

Extend Component for components that need instance methods or fine-grained lifecycle control. PureComponent adds automatic shallow-comparison in shouldComponentUpdate.

components/Counter.js 
import { createElement, Component } from 'flexui';

export class Counter extends Component {
  state = { count: 0 };

  render() {
    return createElement('div', null,
      createElement('p', null, `Count: ${this.state.count}`),
      createElement('button', {
        onClick: () => this.setState(s => ({ count: s.count + 1 }))
      }, 'Increment')
    );
  }
}

Class Lifecycle Methods

MethodWhen it runs
componentDidMount()After first render and DOM insertion
componentDidUpdate(prevProps, prevState)After every re-render
componentWillUnmount()Before component is removed from DOM
shouldComponentUpdate(nextProps, nextState)Return false to skip re-render
static getDerivedStateFromError(error)Error boundary — return new state on child error
componentDidCatch(error, info)Error boundary — side-effect on child error

createElement

Creates a virtual DOM node. Every UI element ultimately calls this function.

createElement(type, props, ...children) → FlexElement
ParamTypeDescription
typestring | Function | ClassHTML tag name or component reference
propsobject | nullAttributes, event handlers, key, ref. Pass null for none.
...childrenFlexElement | string | numberZero or more child nodes
 
// Plain element
createElement('h1', { className: 'title' }, 'Hello');

// Component with children
createElement(Card, { title: 'Welcome' },
  createElement('p', null, 'Card body text.')
);

// Keyed list
items.map(item =>
  createElement('li', { key: item.id }, item.name)
);

Fragment

Groups multiple children without adding an extra DOM node.

import { createElement, Fragment } from 'flexui';

createElement(Fragment, null,
  createElement('dt', null, 'Term'),
  createElement('dd', null, 'Definition')
);

memo

Wraps a functional component and skips re-renders when props are shallowly equal. Pass a custom comparator as the second argument for fine-grained control.

import { memo } from 'flexui';

const Row = memo(function Row({ data }) {
  return createElement('tr', null, ...);
});

// Custom comparator
const StableRow = memo(Row, (prev, next) => prev.data.id === next.data.id);

lazy

Dynamically imports a component on first render. Must be wrapped in <Suspense>.

import { lazy } from 'flexui';
import { Suspense } from 'flexui/components';

const Chart = lazy(() => import('./Chart.js'));

createElement(Suspense, { fallback: createElement('span', null, 'Loading…') },
  createElement(Chart, { data })
);

createContext

Creates a context object. The Provider component broadcasts a value; useContext reads it anywhere in the subtree.

import { createContext } from 'flexui';
import { useContext } from 'flexui/hooks';

const ThemeCtx = createContext('light');

// Provider
createElement(ThemeCtx.Provider, { value: 'dark' }, children);

// Consumer
function ThemedBtn() {
  const theme = useContext(ThemeCtx);
  return createElement('button', { className: `btn--${theme}` }, 'Click');
}

forwardRef

Lets a component receive a ref from its parent and forward it to a DOM node or child component.

import { forwardRef, useRef } from 'flexui';

const FancyInput = forwardRef(function(props, ref) {
  return createElement('input', { ...props, ref, className: 'fancy-input' });
});

// Parent
const inputRef = useRef();
createElement(FancyInput, { ref: inputRef, placeholder: 'Type here' });
inputRef.current.focus(); // direct DOM access
