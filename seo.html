Head Component

Declaratively manage <head> tags from anywhere in the component tree. Works in both CSR and SSR modes. Nested <Head> usage follows a last-write-wins merge strategy.

import { Head } from 'flexui/seo';

export default function AboutPage() {
  return createElement(Fragment, null,
    createElement(Head, null,
      createElement('title', null, 'About — MyApp'),
      createElement('meta', { name: 'description', content: 'Learn about us' }),
      createElement('link', { rel: 'canonical', href: 'https://myapp.com/about' }),
    ),
    createElement('main', null, 'Page content')
  );
}

openGraph

Returns an array of <meta> elements for Open Graph tags. Pass the result inside <Head>.

import { Head, openGraph } from 'flexui/seo';

createElement(Head, null,
  ...openGraph({
    title:       'My Article',
    description: 'A great read',
    image:       'https://myapp.com/og.png',
    url:         'https://myapp.com/blog/my-article',
    type:        'article',
    siteName:    'MyApp',
    locale:      'en_US',
  })
);

import { twitterCard } from 'flexui/seo';

    title:       'My Article',
    description: 'A great read',
    image:       'https://myapp.com/twitter-card.png',
    site:        '@myapp',
    creator:     '@author',
  })
);

jsonLD & Schema Helpers

Generates a <script type="application/ld+json"> tag from a structured data object. Schema helpers pre-fill required fields.

import { jsonLD, articleSchema, breadcrumbSchema } from 'flexui/seo';

createElement(Head, null,
  jsonLD(articleSchema({
    headline:      'My Article',
    datePublished: '2025-03-01',
    author:        { name: 'Jane Doe' },
    image:         'https://myapp.com/article.jpg',
  })),
  jsonLD(breadcrumbSchema([
    { name: 'Home', url: '/' },
    { name: 'Blog', url: '/blog' },
    { name: 'My Article' },
  ]))
);

Sitemap Generation

import { generateSitemap } from 'flexui/seo';

const xml = await generateSitemap({
  baseUrl: 'https://myapp.com',
  routes: [
    { path: '/',      changefreq: 'daily',   priority: 1.0 },
    { path: '/about', changefreq: 'monthly', priority: 0.8 },
    { path: '/blog',  changefreq: 'weekly',  priority: 0.9,
      images: [{ url: 'https://myapp.com/og.png', title: 'OG image' }]
    },
  ],
});

fs.writeFileSync('dist/sitemap.xml', xml);

Or use the CLI:

flexui generate sitemap

robots.txt Generation

import { generateRobots } from 'flexui/seo';

const txt = generateRobots({
  sitemap: 'https://myapp.com/sitemap.xml',
  agents: [
    { agent: '*',          allow: ['/'], disallow: ['/admin'] },
    { agent: 'Googlebot', allow: ['/'] },
  ],
});

Or use the CLI:

flexui generate robots
